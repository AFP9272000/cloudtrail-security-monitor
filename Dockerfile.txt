# Multi-stage build for CloudTrail Security Monitor
# Stage 1: Build stage
FROM python:3.11-slim as builder

# Set working directory
WORKDIR /app

# Install build dependencies
RUN apt-get update && apt-get install -y --no-install-recommends \
    gcc \
    && rm -rf /var/lib/apt/lists/*

# Copy requirements
COPY requirements.txt .

# Install Python dependencies
RUN pip install --user --no-cache-dir --no-warn-script-location -r requirements.txt

# Stage 2: Runtime stage
FROM python:3.11-slim

# Create non-root user for security
RUN groupadd -r cloudtrail && useradd -r -g cloudtrail cloudtrail

# Set working directory
WORKDIR /app

# Copy Python dependencies from builder
COPY --from=builder /root/.local /home/cloudtrail/.local

# Copy application code
COPY src/ ./src/
COPY config/ ./config/
COPY cloudtrail_monitor.py .

# Set proper permissions
RUN chown -R cloudtrail:cloudtrail /app

# Switch to non-root user
USER cloudtrail

# Add local bin to PATH
ENV PATH=/home/cloudtrail/.local/bin:$PATH

# Set Python to run in unbuffered mode
ENV PYTHONUNBUFFERED=1

# Health check
HEALTHCHECK --interval=300s --timeout=30s --start-period=5s --retries=3 \
    CMD python -c "import sys; sys.exit(0)"

# Run the application
ENTRYPOINT ["python", "-u", "cloudtrail_monitor.py"]

# Metadata
LABEL maintainer="your-email@example.com" \
      description="CloudTrail Security Monitor" \
      version="1.0.0"
